Рандомный Цвет

JS:
function getColorRandom(){
const color = `rgb(${Math.round((Math.random()*256))}, ${Math.round((Math.random()*256))}, ${Math.round((Math.random()*256))})`;
return color;
}

Линейный градиент

JS:
function getColorLinear(){
const color = `linear-gradient(${Math.round((Math.random()*256))}0%, ${Math.round((Math.random()*256))}50%, ${Math.round((Math.random()*256))}51%, ${Math.round((Math.random()*256))}100%)`;
return color;
}

Радиальный градиент

JS:
function getColorRadial(){
const color = `radial-gradient(${Math.round((Math.random()*256))}, ${Math.round((Math.random()*256))}, ${Math.round((Math.random()*256))})`;
return color;
}
